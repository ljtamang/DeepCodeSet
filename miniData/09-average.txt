public class Average {
    public static void main(String[] args) {
        
        double[] numArray = { 8, 6, 11, 7};
        double sum = 0.0;

        for (double num: numArray) {
           sum += num;
        }

        double average = sum / numArray.length;
        System.out.format("The average is: %.2f", average);
    }
}
